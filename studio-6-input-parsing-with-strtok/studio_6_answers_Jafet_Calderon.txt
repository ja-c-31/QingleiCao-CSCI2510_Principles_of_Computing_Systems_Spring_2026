#include <stdio.h>
#include <string.h>

int main(void) {
    char input[256];
    char *token;

    int max_args = 15;
    int max_argv_size = max_args + 2; // one for argv[0], one for null terminator
    char *cmd;
    char *my_argv[max_argv_size];

    printf("Enter input: ");

    if (fgets(input, sizeof(input), stdin) == NULL) {
        return 1;
    }

    input[strcspn(input, "\n")] = '\0';

    token = strtok(input, " ");

    cmd = token;

    my_argv[0] = cmd;

    int i = 1;
    while ((token = strtok(NULL, " ")) != NULL && i < max_argv_size - 1) {
        my_argv[i] = token;
        i++;
    }

    my_argv[i] = NULL;

    while (token != NULL) {
        printf("%s\n", token);
        token = strtok(NULL, " ");
    }

    return 0;
}
