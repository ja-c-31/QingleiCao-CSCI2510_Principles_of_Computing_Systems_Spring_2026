#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/wait.h>

int main(void) {
    int fd[2];
    pid_t child1, child2;

    if (pipe(fd) == -1) {
        perror("pipe");
        return 1;
    }

    child1 = fork();
    if (child1 < 0) {
        perror("fork child1");
        return 1;
    }

    if (child1 == 0) {
        if (dup2(fd[1], STDOUT_FILENO) == -1) {
            perror("dup2 child1");
            exit(1);
        }

        char *cmd = "./program1";
        char *myargv[] = {"print", NULL};
        execvp(cmd, myargv);
        perror("Error exec'ing print");
        exit(1);
    }

    child2 = fork();
    if (child2 < 0) {
        perror("fork child2");
        return 1;
    }

    if (child2 == 0) {
        if (dup2(fd[0], STDIN_FILENO) == -1) {
            perror("dup2 child2");
            exit(1);
        }

        char *cmd = "./program2";
        char *myargv[] = {"input", NULL};
        execvp(cmd, myargv);
        perror("Error exec'ing input");
        exit(1);
    }

    waitpid(child1, NULL, 0);
    waitpid(child2, NULL, 0);

    return 0;
}
